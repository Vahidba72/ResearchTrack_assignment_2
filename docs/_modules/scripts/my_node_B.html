#! /usr/bin/env python3
"""
.. module:: node_b

   :synopsis: a service node for the assignment_2_2023 to show the last target


.. moduleauthor:: Vahid Bagherian v.bagherianno@gmail.com

This node gets the last imported target by the user from the launch file and displays it as the output when the service is called.

  
Service:
   -'last_target'

Functions:
   - `put_target`
"""

# Import necessary libraries
import rospy

from assignment_2_2023.srv import LastTarget, LastTargetResponse



[docs]def put_target(req):

	"""
	Callback function of the service `last_target` to extract the last chosen target.
	
	:param req: request for the last target display
	:type req: LastTarget
	:returns: the x and y for the last chosen target
	:rtype: :class:`LastTargetResponse`
	
	
	This function extracts the desired x and y positions ('des_pos_x' and 'des_pos_y') from asssignment1.launch and stores them into response variable of the service (res)
	
	"""

	res = LastTargetResponse()
	
	
	
	res.target_x = rospy.get_param('/des_pos_x',default=0.0)
	res.target_y = rospy.get_param('/des_pos_y',default=0.0)
	
	
	
	return res



[docs]def main():

	"""
	Main function to initialize the node and the service.
	
	This function initializes the node 'node_b_service' and sets up the service 'last_target' with the callback function `put_target`.
	
	"""


	# Initialize the node 
	rospy.init_node("node_b_service")
	rospy.loginfo("target node started and ready to give the last target the user entered")
	
	rospy.Service('last_target', LastTarget, put_target) 
	
	
		
	rospy.spin()

	
	
if __name__ == "__main__":

	a = main()
